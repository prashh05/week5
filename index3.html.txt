  <title>Form Validation</title>

  <!-- Bootstrap CSS -->
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet" />
  <style>
    .d-none {
      display: none !important;
    }
  </style>
</head>
<body class="p-4">

  <div class="container">
    <h3 class="mb-4">User Information Form</h3>
    <form id="userForm" novalidate>
      <!-- Name -->
      <div class="mb-3">
        <label for="name" class="form-label">Name</label>
        <input type="text" class="form-control" id="name" placeholder="Enter your name" required />
        <div id="nameError" class="invalid-feedback d-none">Please enter your name.</div>
      </div>

      <!-- Email -->
      <div class="mb-3">
        <label for="email" class="form-label">Email</label>
        <input type="email" class="form-control" id="email" placeholder="Enter your email" required />
        <div id="emailError" class="invalid-feedback d-none">Please enter a valid email address ending with .com.</div>
      </div>

      <!-- Age -->
      <div class="mb-3">
        <label for="age" class="form-label">Age</label>
        <input type="number" class="form-control" id="age" min="18" max="150" placeholder="Enter your age" required />
        <div id="ageError" class="invalid-feedback d-none">Age must be between 18 and 120.</div>
      </div>

      <button type="submit" class="btn btn-primary">Submit</button>
    </form>

    <!-- Success Message -->
    <div id="successMessage" class="mt-3 alert alert-success d-none">
      <p>Success! Here is the data:</p>
      <p id="successData"></p>
    </div>
  </div>

  <!-- Bootstrap JS -->
  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>

  <script>
    document.getElementById("userForm").addEventListener("submit", function (event) {
      event.preventDefault();

      const nameField = document.getElementById("name");
      const emailField = document.getElementById("email");
      const ageField = document.getElementById("age");

      const nameError = document.getElementById("nameError");
      const emailError = document.getElementById("emailError");
      const ageError = document.getElementById("ageError");

      const successMessage = document.getElementById("successMessage");
      const successData = document.getElementById("successData");

      // Reset errors
      nameError.classList.add("d-none");
      emailError.classList.add("d-none");
      ageError.classList.add("d-none");

      nameField.classList.remove("is-invalid");
      emailField.classList.remove("is-invalid");
      ageField.classList.remove("is-invalid");

      successMessage.classList.add("d-none");

      let isValid = true;

      // Name Validation
      if (!nameField.value.trim()) {
        nameError.classList.remove("d-none");
        nameField.classList.add("is-invalid");
        isValid = false;
      }

      // Email Validation
      const emailPattern = /^[^\s@]+@[^\s@]+\.com$/;
      if (!emailPattern.test(emailField.value)) {
        emailError.classList.remove("d-none");
        emailField.classList.add("is-invalid");
        isValid = false;
      }

      // Age Validation
      const age = parseInt(ageField.value);
      if (isNaN(age) || age < 18 || age > 120) {
        ageError.classList.remove("d-none");
        ageField.classList.add("is-invalid");
        isValid = false;
      }

      // Show Success
      if (isValid) {
        successMessage.classList.remove("d-none");
        successData.innerHTML =
          "<strong>Name:</strong> " + nameField.value + "<br>" +
          "<strong>Email:</strong> " + emailField.value + "<br>" +
          "<strong>Age:</strong> " + ageField.value;
      }
    });
  </script>
